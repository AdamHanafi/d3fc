  var data = fc.utilities.dataGenerator()
    .mu(0.1)                              // Mu value used in the Brownian Motion algorithm
    .sigma(0.1)                           // Sigma value used in the Brownian Motion algorithm
    .startingPrice(100)                   // The initial price for the first generated data point
    .intraDaySteps(50)                    // The amount of steps for each Day, default data is daily
    .fromDate(new Date(2014, 10, 1))      // Generate the data from this date
    .toDate(new Date(2014, 10, 30))       // Generate the data to this date
    .filter(function (date) {             // Specify the function used on the data, this filter will filter out weekends
    	return !(date.getDay() === 0 || date.getDay() === 6);  
    })
    .generate();                          // Generate the data
